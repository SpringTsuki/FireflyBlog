# 1. 刷怪箱的自定义写法

### 生成完整功能的刷怪箱（1.21.5规范）

```json
setblock ~ ~1 ~ minecraft:spawner{
  id: "minecraft:spawner",
  Delay: 0,
  MinSpawnDelay: 200,
  MaxSpawnDelay: 800,
  SpawnCount: 1,
  MaxNearbyEntities: 10,
  RequiredPlayerRange: 16,
  SpawnRange: 4,
  SpawnData: {
    entity: {
      id: "minecraft:zombie",
      DeathLootTable: "your_namespace:entities/custom_zombie",
      Tags: ["custom_monster"]
    }
  }
}
```

```json
# 完整刷怪箱生成指令 (1.21.5)
setblock ~ ~1 ~ minecraft:spawner{
  id: "minecraft:spawner",  # 必须声明方块实体类型
  x: 0, y: 0, z: 0,        # 坐标会自动填充，可省略
  Delay: 0,                 # 当前生成倒计时(刻)，0=立即开始
  MinSpawnDelay: 200,       # 最小生成间隔(刻，默认200=10秒)
  MaxSpawnDelay: 800,       # 最大生成间隔(刻，默认800=40秒)
  SpawnCount: 1,            # 每次生成数量(默认4)
  MaxNearbyEntities: 10,    # 附近同类实体上限(默认6)
  RequiredPlayerRange: 16,   # 玩家激活范围(默认16)
  SpawnRange: 4,            # 生成半径(默认4)
  
  # 单一生成实体配置
  SpawnData: {
    entity: {
      id: "minecraft:zombie",  # 实体ID
      DeathLootTable: "your_namespace:entities/custom_zombie",  # 自定义战利品表
      Tags: ["custom_monster", "special_mob"],  # 识别标签
      
      # 可选实体属性示例
      Health: 30.0f,           # 生命值
      Attributes: [             # 属性修改
        {
          Name: "generic.movement_speed",
          Base: 0.25d
        }
      ],
      HandItems: [              # 手持物品
        {id: "minecraft:iron_sword", Count: 1b},
        {id: "minecraft:shield", Count: 1b}
      ],
      ArmorItems: [             # 装备
        {id: "minecraft:leather_helmet", Count: 1b},
        {id: "minecraft:chainmail_chestplate", Count: 1b},
        {id: "minecraft:iron_leggings", Count: 1b},
        {id: "minecraft:golden_boots", Count: 1b}
      ]
    },
    
    # 自定义生成规则 (覆盖默认光照限制)
    custom_spawn_rules: {
      block_light_limit: {min: 0, max: 15},  # 方块光照限制(0-15)
      sky_light_limit: {min: 0, max: 15}      # 天空光照限制(0-15)
    },
    
    # 装备掉落率设置
    equipment: {
      loot_table: "your_namespace:equipment/zombie_gear",  # 装备战利品表
      slot_drop_chances: {  # 各槽位掉落率(0.0-1.0)
        "mainhand": 0.5f,
        "offhand": 0.3f,
        "head": 0.2f,
        "chest": 0.25f,
        "legs": 0.2f,
        "feet": 0.15f
      }
    }
  },
  
  # 多类型怪物池配置 (与SpawnData二选一)
  SpawnPotentials: [
    # 第一种生成选项
    {
      weight: 3,  # 权重(越高出现概率越大)
      data: {
        entity: {
          id: "minecraft:skeleton",
          DeathLootTable: "your_namespace:entities/custom_skeleton",
          Tags: ["archer"],
          HandItems: [{id: "minecraft:bow", Count: 1b}]
        },
        custom_spawn_rules: {
          block_light_limit: {min: 0, max: 7},
          sky_light_limit: {min: 0, max: 7}
        }
      }
    },
    
    # 第二种生成选项
    {
      weight: 1,
      data: {
        entity: {
          id: "minecraft:spider",
          DeathLootTable: "your_namespace:entities/custom_spider",
          Tags: ["poisonous"],
          ActiveEffects: [
            {
              Id: 19b,  # 中毒效果
              Amplifier: 1b,
              Duration: 200
            }
          ]
        }
      }
    }
  ]
}
```


这！是一个最基本的刷怪箱！它会刷新僵尸！
